import { test, expect } from '@playwright/test';

test.describe('About & Stats modals', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/');
  });

  test('About modal opens and closes', async ({ page }) => {
    await page.locator('#aboutBtn').click();
    await expect(page.locator('[data-testid="about-modal"]')).toBeVisible();
    await page.locator('[data-testid="modal-close"]').click();
    await expect(page.locator('[data-testid="about-modal"]')).toHaveCount(0);
  });

  test('Stats modal opens and closes', async ({ page }) => {
    await page.locator('#usageStatsBtn').click();
    await expect(page.locator('[data-testid="stats-modal"]')).toBeVisible();
    await page.locator('[data-testid="modal-close"]').click();
    await expect(page.locator('[data-testid="stats-modal"]')).toHaveCount(0);
  });
});
