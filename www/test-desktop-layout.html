Search Interface Test
Console Commands

Run these commands in the browser console to test the layout:

// Test search row layout
const searchRow = document.querySelector('.search-row');
if (searchRow) {
    const styles = getComputedStyle(searchRow);
    console.log('Search Row:', {
        display: styles.display,
        flexDirection: styles.flexDirection,
        gap: styles.gap
    });
}

// Test genre filter width
const genreFilter = document.querySelector('.genre-filter');
if (genreFilter) {
    const styles = getComputedStyle(genreFilter);
    console.log('Genre Filter:', {
        width: styles.width,
        minWidth: styles.minWidth,
        maxWidth: styles.maxWidth
    });
}

// Test search input
const searchInput = document.querySelector('.search-input');
if (searchInput) {
    const styles = getComputedStyle(searchInput);
    console.log('Search Input:', {
        width: styles.width,
        flex: styles.flex,
        minWidth: styles.minWidth
    });
}

// Test buttons
const searchBtn = document.querySelector('.search-btn');
const clearBtn = document.querySelector('.clear-search-btn');
if (searchBtn && clearBtn) {
    const searchStyles = getComputedStyle(searchBtn);
    const clearStyles = getComputedStyle(clearBtn);
    console.log('Search Button:', {
        width: searchStyles.width,
        minWidth: searchStyles.minWidth,
        flex: searchStyles.flex
    });
    console.log('Clear Button:', {
        width: clearStyles.width,
        minWidth: clearStyles.minWidth,
        flex: clearStyles.flex
    });
}
